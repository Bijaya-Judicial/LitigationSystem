<script>
let chart;

function loadChart(type) {

    let labels = [];
    let data = [];

    if (type === 'court') {
        labels = {{ court_labels|tojson }};
        data = {{ court_data|tojson }};
    }

    if (type === 'case_type') {
        labels = {{ type_labels|tojson }};
        data = {{ type_data|tojson }};
    }

    if (type === 'office') {
        labels = {{ office_labels|tojson }};
        data = {{ office_data|tojson }};
    }

    if (type === 'year') {
        labels = {{ year_labels|tojson }};
        data = {{ year_data|tojson }};
    }

    if (chart) chart.destroy();

    chart = new Chart(document.getElementById('barChart'), {
        type: 'bar',
        data: {
            labels: labels,
            datasets: [{
                label: 'Cases',
                data: data,
                backgroundColor: '#0d6efd'
            }]
        },
        options: {
            onClick: function(evt, elements) {
                if (elements.length > 0) {
                    let index = elements[0].index;
                    let value = labels[index];
                    window.location.href = `/dashboard?${type}=` + value;
                }
            }
        }
    });
}

loadChart('court');

new Chart(document.getElementById('pieChart'), {

        datasets: [{

        }]
    }
});
